477/*
478 * Auxiliary function to add an outgoing edge to a vertex (this),
479 * with a given destination vertex (d) and edge weight (w).
480 */
481template<class T>
482Edge<T> *Vertex<T>::addEdge(Vertex<T> *d, double w) {
483 auto newEdge = new Edge<T>(this, d, w);
484 adj.push_back(newEdge);
485 d->incoming.push_back(newEdge);
486 return newEdge;
487}
488
489/*
490 * Auxiliary function to remove an outgoing edge (with a given destination (d))
491 * from a vertex (this).
492 * Returns true if successful, and false if such edge does not exist.
493 */
494template<class T>
495bool Vertex<T>::removeEdge(T in) {
496 bool removedEdge = false;
497 auto it = adj.begin();
498 while (it != adj.end()) {
499 Edge<T> *edge = *it;
500 Vertex<T> *dest = edge->getDest();
501 if (dest->getInfo() == in) {
502 it = adj.erase(it);
503 deleteEdge(edge);
504 removedEdge = true; // allows for multiple edges to connect the same pair of vertices (multigraph)
505 } else {
506 it++;
507 }
508 }
509 return removedEdge;
510}

512/*
513 * Auxiliary function to remove an outgoing edge of a vertex.
514 */
515template<class T>
516void Vertex<T>::removeOutgoingEdges() {
517 auto it = adj.begin();
518 while (it != adj.end()) {
519 Edge<T> *edge = *it;
520 it = adj.erase(it);
521 deleteEdge(edge);
522 }
523}

620template<class T>
621void Vertex<T>::deleteEdge(Edge<T> *edge) {
622 Vertex<T> *dest = edge->getDest();
623 // Remove the corresponding edge from the incoming list
624 auto it = dest->incoming.begin();
625 while (it != dest->incoming.end()) {
626 if ((*it)->getOrig()->getInfo() == info) {
627 it = dest->incoming.erase(it);
628 } else {
629 it++;
630 }
631 }
632 delete edge;
633}

754/*
755 * Removes a vertex with a given content (in) from a graph (this), and
756 * all outgoing and incoming edges.
757 * Returns true if successful, and false if such vertex does not exist.
758 */
759template<class T>
760bool Graph<T>::removeVertex(const T &in) {
761 for (auto it = vertexSet.begin(); it != vertexSet.end(); it++) {
762 if ((*it)->getInfo() == in) {
763 auto v = *it;
764 v->removeOutgoingEdges();
765 for (auto u: vertexSet) {
766 u->removeEdge(v->getInfo());
767 }
768 vertexSet.erase(it);
769 delete v;
770 return true;
771 }
772 }
773 return false;
774}

776/*
777 * Adds an edge to a graph (this), given the contents of the source and
778 * destination vertices and the edge weight (w).
779 * Returns true if successful, and false if the source or destination vertex does not exist.

781template<class T>
782bool Graph<T>::addEdge(const T &sourc, const T &dest, double w) {
783 auto v1 = findVertex(sourc);
784 auto v2 = findVertex(dest);
785 if (v1 == nullptr || v2 == nullptr)
786 return false;
787 v1->addEdge(v2, w);
788 return true;

791/*
792 * Removes an edge from a graph (this).
793 * The edge is identified by the source (sourc) and destination (dest) contents.
794 * Returns true if successful, and false if such edge does not exist.
795 */
796template<class T>
797bool Graph<T>::removeEdge(const T &sourc, const T &dest) {
798 Vertex<T> *srcVertex = findVertex(sourc);
799 if (srcVertex == nullptr) {
800 return false;
801 }
802 return srcVertex->removeEdge(dest);

805template<class T>
806bool Graph<T>::addBidirectionalEdge(const T &sourc, const T &dest, double w) {
807 auto v1 = findVertex(sourc);
808 auto v2 = findVertex(dest);
809 if (v1 == nullptr || v2 == nullptr)
810 return false;
811 auto e1 = v1->addEdge(v2, w);
812 auto e2 = v2->addEdge(v1, w);
813 e1->setReverse(e2);
814 e2->setReverse(e1);
815 return true;
816}
