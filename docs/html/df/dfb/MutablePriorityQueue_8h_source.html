157template <class T>
158void MutablePriorityQueue<T>::heapifyUp(unsigned i) {
159 auto x = H[i];
160 while (i > 1 && *x < *H[parent(i)]) {
161 set(i, H[parent(i)]);
162 i = parent(i);
163 }
164 set(i, x);
165}

176template <class T>
177void MutablePriorityQueue<T>::heapifyDown(unsigned i) {
178 auto x = H[i];
179 while (true) {
180 unsigned k = leftChild(i);
181 if (k >= H.size())
182 break;
183 if (k+1 < H.size() && *H[k+1] < *H[k])
184 ++k; // right child of i
185 if ( ! (*H[k] < *x) )
186 break;
187 set(i, H[k]);
188 i = k;
189 }
190 set(i, x);
191}
